# ATH Protocol Client Reference Implementation
## 客户端核心实现规范
本文档提供ATH协议客户端的标准参考实现，所有兼容ATH协议的客户端都应该遵循此实现规范。
### 核心类结构
```python
import requests
import json
import hashlib
import hmac
from typing import Dict, List, Optional
class ATHClient:
    """
    ATH协议客户端核心实现类
    负责与服务端完成8步握手流程，管理访问凭证，发起业务请求
    """
    
    def __init__(self, client_id: str, client_secret: str, server_url: str):
        """
        初始化ATH客户端
        :param client_id: 客户端唯一标识
        :param client_secret: 客户端密钥，用于签名验证
        :param server_url: 服务端ATH接口地址
        """
        self.client_id = client_id
        self.client_secret = client_secret.encode('utf-8')
        self.server_url = server_url.rstrip('/')
        self.access_token: Optional[str] = None
        self.token_expire_at: Optional[int] = None
        self.granted_scopes: List[str] = []
    def _generate_signature(self, data: Dict) -> str:
        """
        生成请求签名，防止请求被篡改
        :param data: 待签名的请求数据
        :return: HMAC-SHA256签名
        """
        sorted_data = json.dumps(data, sort_keys=True).encode('utf-8')
        return hmac.new(self.client_secret, sorted_data, hashlib.sha256).hexdigest()
    def step1_identity_announcement(self) -> Dict:
        """
        步骤1: 客户端身份宣告
        向服务端发送客户端基本信息，发起握手流程
        :return: 服务端身份回应数据（步骤2结果）
        """
        data = {
            "client_id": self.client_id,
            "client_type": "standard_ath_client",
            "version": "0.1.0",
            "supported_algorithms": ["HS256", "RS256"],
            "timestamp": int(__import__('time').time())
        }
        data["signature"] = self._generate_signature(data)
        response = requests.post(f"{self.server_url}/ath/v1/announce", json=data)
        response.raise_for_status()
        return response.json()
    def step3_scope_request(self, requested_scopes: List[str], user_context: Optional[Dict] = None) -> Dict:
        """
        步骤3: 作用域请求
        向服务端请求需要的访问权限
        :param requested_scopes: 请求的权限列表
        :param user_context: 用户上下文信息（可选）
        :return: 服务端作用域协商结果（步骤4结果）
        """
        data = {
            "client_id": self.client_id,
            "requested_scopes": requested_scopes,
            "user_context": user_context or {},
            "timestamp": int(__import__('time').time())
        }
        data["signature"] = self._generate_signature(data)
        response = requests.post(f"{self.server_url}/ath/v1/scope-request", json=data)
        response.raise_for_status()
        return response.json()
    def step5_identity_verification(self, verification_data: Dict) -> Dict:
        """
        步骤5: 身份验证
        提交服务端要求的验证材料，完成身份校验
        :param verification_data: 验证材料（根据步骤4的要求提供）
        :return: 服务端颁发的凭证信息（步骤6结果）
        """
        data = {
            "client_id": self.client_id,
            "verification_data": verification_data,
            "timestamp": int(__import__('time').time())
        }
        data["signature"] = self._generate_signature(data)
        response = requests.post(f"{self.server_url}/ath/v1/verify", json=data)
        response.raise_for_status()
        credential = response.json()
        self.access_token = credential["access_token"]
        self.token_expire_at = credential["expires_at"]
        self.granted_scopes = credential["scopes"]
        return credential
    def step7_credential_confirmation(self) -> bool:
        """
        步骤7: 凭证确认
        确认收到的凭证有效，完成握手流程
        :return: 确认是否成功
        """
        data = {
            "client_id": self.client_id,
            "access_token": self.access_token,
            "timestamp": int(__import__('time').time())
        }
        data["signature"] = self._generate_signature(data)
        response = requests.post(f"{self.server_url}/ath/v1/confirm", json=data)
        return response.status_code == 200
    def execute_handshake(self, requested_scopes: List[str], verification_provider) -> bool:
        """
        一键执行完整8步握手流程
        :param requested_scopes: 请求的权限列表
        :param verification_provider: 验证材料提供函数，接收协商结果返回验证数据
        :return: 握手是否成功
        """
        try:
            # 步骤1-2: 身份宣告与回应
            server_info = self.step1_identity_announcement()
            
            # 步骤3-4: 作用域请求与协商
            scope_requirements = self.step3_scope_request(requested_scopes)
            
            # 步骤5-6: 身份验证与凭证颁发
            verification_data = verification_provider(scope_requirements)
            self.step5_identity_verification(verification_data)
            
            # 步骤7-8: 凭证确认与完成
            return self.step7_credential_confirmation()
        except Exception as e:
            print(f"握手失败: {str(e)}")
            return False
    def request(self, path: str, method: str = "GET", **kwargs) -> Dict:
        """
        发起业务请求（握手成功后使用）
        :param path: 请求路径
        :param method: HTTP方法
        :param kwargs: 请求参数
        :return: 响应数据
        """
        if not self.access_token:
            raise RuntimeError("请先完成ATH握手流程")
        
        headers = kwargs.pop("headers", {})
        headers["Authorization"] = f"Bearer {self.access_token}"
        
        url = f"{self.server_url}{path}"
        response = requests.request(method, url, headers=headers, **kwargs)
        response.raise_for_status()
        return response.json()
```
### 使用示例
```python
# 初始化客户端
client = ATHClient(
    client_id="your_client_id",
    client_secret="your_client_secret",
    server_url="https://api.example.com"
)
# 定义验证材料提供函数
def provide_verification(requirements):
    # 根据服务端的要求返回对应的验证数据
    return {
        "user_token": "user_auth_token",
        "additional_claims": {"role": "user"}
    }
# 执行握手
if client.execute_handshake(
    requested_scopes=["user:read", "data:write"],
    verification_provider=provide_verification
):
    print("✅ 握手成功！")
    # 发起业务请求
    user_info = client.request("/user/info", method="GET")
    print(f"用户信息: {user_info}")
else:
    print("❌ 握手失败！")
```
### 实现规范
1. 所有请求必须包含签名，防止篡改
2. 凭证需要妥善保管，避免泄露
3. 凭证过期前需要主动刷新
4. 错误处理需要覆盖所有可能的异常情况
5. 需要支持多种签名算法以兼容不同服务端
