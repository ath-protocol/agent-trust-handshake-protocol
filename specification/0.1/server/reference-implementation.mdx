# ATH Protocol Server Reference Implementation
## 服务端核心实现规范
本文档提供ATH协议服务端的标准参考实现，所有兼容ATH协议的服务端都应该遵循此实现规范。
### 核心类结构
```python
from flask import Flask, request, jsonify
import jwt
import datetime
import hmac
import hashlib
import json
import requests
from typing import Dict, List, Callable
app = Flask(__name__)
class ATHServer:
    """
    ATH协议服务端核心实现类
    负责处理客户端的12步握手请求，验证身份，颁发凭证，校验业务请求
    """
    
    def __init__(self, server_did: str, server_secret: str, registry_url: str, jwt_secret: str):
        """
        初始化ATH服务端
        :param server_did: 服务端去中心化身份标识
        :param server_secret: 服务端密钥
        :param registry_url: 身份注册中心地址
        :param jwt_secret: JWT签名密钥
        """
        self.server_did = server_did
        self.server_secret = server_secret.encode('utf-8')
        self.registry_url = registry_url.rstrip('/')
        self.jwt_secret = jwt_secret
        self.client_registry: Dict[str, str] = {}  # 客户端DID -> 客户端密钥映射
        self.permission_policies: Dict[str, Callable] = {}  # 权限策略函数
        self.session_manager: Dict[str, Dict] = {}  # 会话管理
    def _generate_signature(self, data: Dict) -> str:
        """
        生成响应签名
        :param data: 待签名数据
        :return: 签名
        """
        sorted_data = json.dumps(data, sort_keys=True).encode('utf-8')
        return hmac.new(self.server_secret, sorted_data, hashlib.sha256).hexdigest()
    def _verify_client_signature(self, client_did: str, data: Dict, signature: str) -> bool:
        """
        验证客户端请求签名
        :param client_did: 客户端DID
        :param data: 请求数据
        :param signature: 客户端签名
        :return: 签名是否有效
        """
        if client_did not in self.client_registry:
            return False
        
        client_secret = self.client_registry[client_did].encode('utf-8')
        data.pop("signature", None)
        sorted_data = json.dumps(data, sort_keys=True).encode('utf-8')
        expected_signature = hmac.new(client_secret, sorted_data, hashlib.sha256).hexdigest()
        return hmac.compare_digest(expected_signature, signature)
    def _verify_client_identity(self, client_did: str) -> bool:
        """
        向身份注册中心验证客户端身份
        :param client_did: 客户端DID
        :return: 身份是否有效
        """
        data = {
            "did": client_did,
            "verifier_did": self.server_did,
            "timestamp": int(datetime.datetime.now().timestamp())
        }
        data["signature"] = self._generate_signature(data)
        response = requests.post(f"{self.registry_url}/verify", json=data)
        return response.status_code == 200 and response.json().get("valid", False)
    def register_client(self, client_did: str, client_secret: str) -> None:
        """
        注册合法客户端
        :param client_did: 客户端DID
        :param client_secret: 客户端密钥
        """
        self.client_registry[client_did] = client_secret
    def register_permission_policy(self, scope: str, policy: Callable) -> None:
        """
        注册权限策略
        :param scope: 权限名称
        :param policy: 策略函数，接收客户端信息返回是否允许
        """
        self.permission_policies[scope] = policy
    def handle_step1_handshake_request(self) -> Dict:
        """
        处理步骤1: 客户端握手请求
        :return: 步骤2: 服务端握手响应
        """
        print("【步骤2】返回握手响应")
        data = request.json
        client_did = data.get("client_did")
        signature = data.get("signature")
        
        if not self._verify_client_signature(client_did, data.copy(), signature):
            return {"error": "无效的签名"}, 401
        
        # 步骤2: 返回服务端身份信息
        response = {
            "server_did": self.server_did,
            "capabilities": ["api_server", "secure_communication"],
            "supported_algorithms": ["HS256", "RS256", "TLS1.3"],
            "timestamp": int(datetime.datetime.now().timestamp())
        }
        response["signature"] = self._generate_signature(response)
        return jsonify(response), 200
    def handle_step3_verify_client(self) -> bool:
        """
        步骤3: 验证客户端身份（内部处理）
        :return: 身份是否有效
        """
        data = request.json
        client_did = data.get("client_did")
        print("【步骤3-4】验证客户端身份")
        return self._verify_client_identity(client_did)
    def handle_step7_permission_request(self) -> Dict:
        """
        处理步骤7: 权限请求
        :return: 步骤8: 权限审批结果
        """
        print("【步骤8】返回权限审批结果")
        data = request.json
        client_did = data.get("client_did")
        requested_scopes = data.get("requested_scopes", [])
        signature = data.get("signature")
        
        if not self._verify_client_signature(client_did, data.copy(), signature):
            return {"error": "无效的签名"}, 401
        
        # 执行权限审批
        approved_scopes = []
        for scope in requested_scopes:
            if scope in self.permission_policies:
                if self.permission_policies[scope](data.get("user_context", {})):
                    approved_scopes.append(scope)
        
        response = {
            "approved_scopes": approved_scopes,
            "expires_in": 1800,
            "timestamp": int(datetime.datetime.now().timestamp())
        }
        response["signature"] = self._generate_signature(response)
        return jsonify(response), 200
    def handle_step9_permission_confirm(self) -> Dict:
        """
        处理步骤9: 权限确认
        :return: 确认结果
        """
        print("【步骤9】权限确认完成")
        data = request.json
        client_did = data.get("client_did")
        signature = data.get("signature")
        
        if not self._verify_client_signature(client_did, data.copy(), signature):
            return {"error": "无效的签名"}, 401
        
        return jsonify({"status": "success"}), 200
    def handle_step10_key_negotiation(self) -> Dict:
        """
        处理步骤10: 密钥协商请求
        :return: 步骤11: 密钥协商响应
        """
        print("【步骤11】返回密钥协商响应")
        data = request.json
        client_did = data.get("client_did")
        signature = data.get("signature")
        
        if not self._verify_client_signature(client_did, data.copy(), signature):
            return {"error": "无效的签名"}, 401
        
        # 简化密钥协商实现
        key_response = {
            "key_params": {
                "key_exchange": "ECDH",
                "curve": "secp256r1",
                "server_public_key": "server_pub_key_xyz"
            },
            "timestamp": int(datetime.datetime.now().timestamp())
        }
        key_response["signature"] = self._generate_signature(key_response)
        return jsonify(key_response), 200
    def handle_step12_session_establish(self) -> Dict:
        """
        处理步骤12: 会话建立
        :return: 会话建立结果
        """
        print("【步骤12】会话建立完成")
        data = request.json
        client_did = data.get("client_did")
        
        # 生成访问凭证
        access_token = jwt.encode({
            "client_did": client_did,
            "scopes": data.get("confirmed_scopes", []),
            "exp": datetime.datetime.utcnow() + datetime.timedelta(minutes=30),
            "iat": datetime.datetime.utcnow()
        }, self.jwt_secret, algorithm="HS256")
        
        # 记录会话
        self.session_manager[client_did] = {
            "access_token": access_token,
            "created_at": datetime.datetime.now().timestamp(),
            "expires_at": datetime.datetime.now().timestamp() + 1800
        }
        
        return jsonify({
            "status": "success",
            "access_token": access_token,
            "expires_in": 1800
        }), 200
    def verify_request(self, required_scopes: List[str] = None) -> Dict:
        """
        验证业务请求的凭证和权限
        :param required_scopes: 需要的权限列表
        :return: 解码后的凭证payload
        """
        auth_header = request.headers.get("Authorization")
        if not auth_header or not auth_header.startswith("Bearer "):
            return {"error": "需要ATH认证"}, 401
        
        token = auth_header.split(" ")[1]
        try:
            payload = jwt.decode(token, self.jwt_secret, algorithms=["HS256"])
        except jwt.InvalidTokenError:
            return {"error": "无效的凭证"}, 401
        
        # 验证权限
        if required_scopes:
            client_scopes = payload.get("scopes", [])
            for scope in required_scopes:
                if scope not in client_scopes:
                    return {"error": f"缺少权限: {scope}"}, 403
        
        return payload
```
### 服务端使用示例
```python
# 初始化ATH服务端
ath_server = ATHServer(
    server_did="did:ath:server:001",
    server_secret="your_server_secret",
    registry_url="https://registry.example.com",
    jwt_secret="your_jwt_secret"
)
# 注册合法客户端
ath_server.register_client(
    client_did="did:ath:agent:001",
    client_secret="client_secret_abc123"
)
# 注册权限策略
ath_server.register_permission_policy("order:create", lambda ctx: ctx.get("user_level") >= 1)
ath_server.register_permission_policy("payment:process", lambda ctx: ctx.get("balance") >= 100)
# 注册ATH协议路由
@app.route('/ath/v1/handshake/request', methods=['POST'])
def step1():
    return ath_server.handle_step1_handshake_request()
@app.route('/ath/v1/permission/request', methods=['POST'])
def step7():
    return ath_server.handle_step7_permission_request()
@app.route('/ath/v1/permission/confirm', methods=['POST'])
def step9():
    return ath_server.handle_step9_permission_confirm()
@app.route('/ath/v1/session/negotiate', methods=['POST'])
def step10():
    return ath_server.handle_step10_key_negotiation()
# 业务接口示例
@app.route('/order/create', methods=['POST'])
def create_order():
    # 验证请求需要order:create权限
    payload = ath_server.verify_request(required_scopes=["order:create"])
    if isinstance(payload, tuple) and payload[1] != 200:
        return payload
    
    # 执行业务逻辑
    return jsonify({
        "order_id": "order_" + str(datetime.datetime.now().timestamp()),
        "status": "success",
        "message": "订单创建成功"
    })
if __name__ == "__main__":
    print("🚀 ATH服务端启动，端口5000")
    app.run(port=5000, debug=False)
```
### 实现规范
1. 必须支持DID身份体系和身份注册中心验证
2. 必须验证所有请求的签名，防止伪造请求
3. 必须实现完整的12步握手流程
4. 权限控制必须遵循最小权限原则
5. 必须实现会话管理和凭证过期机制
6. 需要记录完整的审计日志
